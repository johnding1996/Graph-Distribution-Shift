14# Add support for an external "COM Extensions" path. 

15# Concept is that you can register a seperate path to be used for 

16# COM extensions, outside of the win32com directory. These modules, however, 

17# look identical to win32com built-in modules. 

18# This is the technique that we use for the "standard" COM extensions. 

19# eg "win32com.mapi" or "win32com.axscript" both work, even though they do not 

20# live under the main win32com directory. 

21__gen_path__ = '' 

22__build_path__ = None 

23### TODO - Load _all_ \\Extensions subkeys - for now, we only read the default 

24### Modules will work if loaded into "win32comext" path.

26def SetupEnvironment(): 

27 HKEY_LOCAL_MACHINE = -2147483646 # Avoid pulling in win32con for just these... 

28 KEY_QUERY_VALUE = 0x1 

29 # Open the root key once, as this is quite slow on NT. 

30 try: 

31 keyName = "SOFTWARE\\Python\\PythonCore\\%s\\PythonPath\\win32com" % sys.winver 

32 key = win32api.RegOpenKey(HKEY_LOCAL_MACHINE , keyName, 0, KEY_QUERY_VALUE) 

33 except (win32api.error, AttributeError): 

34 key = None 

35 

36 try: 

37 found = 0 

38 if key is not None: 

39 try: 

40 __path__.append( win32api.RegQueryValue(key, "Extensions" )) 

41 found = 1 

42 except win32api.error: 

43 # Nothing registered 

44 pass 

45 if not found: 

46 try: 

47 __path__.append( win32api.GetFullPathName( __path__[0] + "\\..\\win32comext") ) 

48 except win32api.error: 

49 # Give up in disgust! 

50 pass 

51 

52 # For the sake of developers, we also look up a "BuildPath" key 

53 # If extension modules add support, we can load their .pyd's from a completely 

54 # different directory (see the comments below) 

55 try: 

56 if key is not None: 

57 global __build_path__ 

58 __build_path__ = win32api.RegQueryValue(key, "BuildPath") 

59 __path__.append(__build_path__) 

60 except win32api.error: 

61 # __build_path__ neednt be defined. 

62 pass 

63 global __gen_path__ 

64 if key is not None: 

65 try: 

66 __gen_path__ = win32api.RegQueryValue(key, "GenPath") 

67 except win32api.error: 

68 pass 

69 finally: 

70 if key is not None: 

71 key.Close()
