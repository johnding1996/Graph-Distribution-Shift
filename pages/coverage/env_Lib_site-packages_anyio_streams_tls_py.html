64 

65 This performs a TLS handshake with the peer. 

66 

67 :param transport_stream: a bytes-transporting stream to wrap 

68 :param server_side: ``True`` if this is the server side of the connection, ``False`` if

71 context has not been provided. 

72 :param hostname: host name of the peer (if host name checking is desired) 

73 :param ssl_context: the SSLContext object to use (if not provided, a secure default will be 

74 created) 

75 :param standard_compatible: if ``False``, skip the closing handshake when closing the 

76 connection, and don't raise an exception if the peer does the same 

77 :raises ~ssl.SSLError: if the TLS handshake fails 

78 

79 """ 

80 if server_side is None: 

81 server_side = not hostname 

82 

83 if not ssl_context: 

84 purpose = ssl.Purpose.CLIENT_AUTH if server_side else ssl.Purpose.SERVER_AUTH 

85 ssl_context = ssl.create_default_context(purpose) 

86 

87 bio_in = ssl.MemoryBIO() 

88 bio_out = ssl.MemoryBIO() 

89 ssl_object = ssl_context.wrap_bio(bio_in, bio_out, server_side=server_side, 

90 server_hostname=hostname)

92 standard_compatible=standard_compatible, _ssl_object=ssl_object,

102 result = func(*args) 

103 except ssl.SSLWantReadError: 

104 try: 

105 # Flush any pending writes first

121 raise BrokenResourceError from exc
