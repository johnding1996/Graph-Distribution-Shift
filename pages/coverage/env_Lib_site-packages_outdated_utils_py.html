36@contextmanager 

37def cache_file(package, mode): 

38 """ 

39 Yields a file-like object for the purpose of writing to or 

40 reading from the cache. 

41 

42 The code: 

43 

44 with cache_file(...) as f: 

45 # do stuff with f 

46 

47 is guaranteed to convert any exceptions to warnings (*), 

48 both in the cache_file(...) call and the 'do stuff with f' 

49 block. 

50 

51 The file is automatically closed upon exiting the with block. 

52 

53 If getting an actual file fails, yields a DummyFile. 

54 

55 :param package: the name of the package being checked as a string 

56 :param mode: the mode to open the file in, either 'r' or 'w' 

57 """ 

58 

59 f = DummyFile() 

60 

61 # We have to wrap the whole function body in this block to guarantee 

62 # catching all exceptions. In particular the yield needs to be inside 

63 # to catch exceptions coming from the with block. 

64 with exception_to_warning('use cache while checking for outdated package', 

65 OutdatedCacheFailedWarning): 

66 try: 

67 cache_path = os.path.join(tempfile.gettempdir(), 

68 get_cache_filename(package)) 

69 if mode == 'w' or os.path.exists(cache_path): 

70 f = open(cache_path, mode) 

71 finally: 

72 # Putting the yield in the finally section ensures that exactly 

73 # one thing is yielded once, otherwise @contextmanager would 

74 # raise an exception. 

75 with f: # closes the file afterards 

76 yield f
